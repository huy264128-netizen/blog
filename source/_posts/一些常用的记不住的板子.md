## manacher (回文串)
```cpp
vector<pair<char, int>> manacher(vector<char> rawData, tuple<char, char, char> unused_char = {'#', '^', '?'}) { // 如果有冲突请使用自定义的unused_char
    auto ans = vector<pair<char, int>>(rawData.size() * 2 + 1);
    ans[0].first = get<1>(unused_char);
    for (int i = 0; i < rawData.size(); i++) {
        ans[i * 2 + 1].first = rawData[i];
        ans[i * 2 + 2].first = get<0>(unused_char);
    }
    ans[rawData.size() * 2].first = get<2>(unused_char);
    int center = 0;
    int rightMax = 0;
    for (int i = 1; i < rawData.size() * 2; i++) {
        if (i > rightMax) {
            int v = 1;
            for (; ans[i - v].first == ans[i + v].first; v++);
            v--;
            ans[i].second = v;
            center = i;
            rightMax = i + v;
        } else {
            if (ans[center * 2 - i].second < rightMax - i) {
                ans[i].second = ans[2 * center - i].second;
            } else {
                int v = rightMax - i;
                for (; ans[i - v].first == ans[i + v].first && i + v < ans.size() && i > v; v++);
                v--;
                ans[i].second = v;
                center = i;
                rightMax = i + v;
            }
        }
    }
    return ans;
}
```

## 线段树

```cpp
template<typename T>
struct node{
    T value;
    T cache;
    int l;
    int r;
};
template <typename T>
class segment_tree {
protected:
    int length;
    node<T>* data;
    T create_tree(int p,int l,int r,T* arr) {
        data[p].l=l;
        data[p].r=r;
        data[p].cache=T();
        if (l==r) {
            data[p].value=arr[l];
        }else {
            data[p].value=create_tree(p*2,l,(l+r)/2,arr)+create_tree(p*2+1,(l+r)/2+1,r,arr);
        }
        return data[p].value;
    }
public:
    explicit segment_tree(T* arr, const int l){
        length=l;
        data=new node<T>[l*4+4];
        create_tree(1,1,length,arr);
    }
    ~segment_tree() {
        delete data;
    }
    T get_sum(int l,int r,int p=1) {
        T sum=T();
        if (data[p].l>r||data[p].r<l) {
            return sum; //
        }
        if (data[p].l>=l&&data[p].r<=r) {
            return data[p].value; //
        }
        if (data[p].cache!=T()) {
            if (data[p*2].r!=data[p*2].l) data[p*2].cache+=data[p].cache;
            data[p*2].value+=(data[p*2].r-data[p*2].l+1)*data[p].cache;
            if (data[p*2+1].r!=data[p*2+1].l) data[p*2+1].cache+=data[p].cache;
            data[p*2+1].value+=(data[p*2+1].r-data[p*2+1].l+1)*data[p].cache;
            data[p].cache=T();
        }
        sum=get_sum(l,r,p*2)+get_sum(l,r,p*2+1);
        return sum;
    }
    void add(int l,int r,T value,int p=1) {
        if (data[p].l>r||data[p].r<l) return;
        if (data[p].l>=l&&data[p].r<=r) {
            data[p].cache+=value;
            data[p].value+=(data[p].r-data[p].l+1)*value;
            return;//
        }
        data[p].value+=(min(data[p].r,r)-max(data[p].l,l)+1)*value;
        add(l,r,value,p*2);
        add(l,r,value,p*2+1);
    }
};
```

$$A_n^m = \frac{n!}{(n-m)!} = n(n-1)(n-2)\dots(n-m+1)$$

$$C_n^m = \binom{n}{m} = \frac{A_n^m}{m!} = \frac{n!}{m!(n-m)!}$$

```
/**
 * @brief 快速幂运算 (a^b % mod)
 * @param a 底数
 * @param b 指数
 * @param m 模数
 * @return long long 结果
 */
long long fastPow(long long a, long long b, long long m) {
    long long res = 1;
    a %= m; // 防止 a 大于 m 的情况
    while (b > 0) {
        if (b & 1) res = (__int128)res * a % m; // 使用 __int128 防止 res * a 溢出
        a = (__int128)a * a % m;
        b >>= 1;
    }
    return res % m;
}
```
