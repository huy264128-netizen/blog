## manacher (回文串)
```cpp
vector<pair<char, int>> manacher(vector<char> rawData, tuple<char, char, char> unused_char = {'#', '^', '?'}) { // 如果有冲突请使用自定义的unused_char
    auto ans = vector<pair<char, int>>(rawData.size() * 2 + 1);
    ans[0].first = get<1>(unused_char);
    for (int i = 0; i < rawData.size(); i++) {
        ans[i * 2 + 1].first = rawData[i];
        ans[i * 2 + 2].first = get<0>(unused_char);
    }
    ans[rawData.size() * 2].first = get<2>(unused_char);
    int center = 0;
    int rightMax = 0;
    for (int i = 1; i < rawData.size() * 2; i++) {
        if (i > rightMax) {
            int v = 1;
            for (; ans[i - v].first == ans[i + v].first; v++);
            v--;
            ans[i].second = v;
            center = i;
            rightMax = i + v;
        } else {
            if (ans[center * 2 - i].second < rightMax - i) {
                ans[i].second = ans[2 * center - i].second;
            } else {
                int v = rightMax - i;
                for (; ans[i - v].first == ans[i + v].first && i + v < ans.size() && i > v; v++);
                v--;
                ans[i].second = v;
                center = i;
                rightMax = i + v;
            }
        }
    }
    return ans;
}
```
